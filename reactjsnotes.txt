(12)FILTER=>
//////filter//////
> filter help to filter out the way
> it only return those value for which condition/output is true
> it may or may not return same length of output

var a = [4,7,9,12,3,45,67,8,18,31]
a.filter((data) => {return data>20})
[45, 67, 31]

[-2, 0, 2, 4, 6]

[-1, 1, 2, 3]
